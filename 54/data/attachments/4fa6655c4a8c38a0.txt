var jsonData = pm.response.json();
pm.test("Status code name has string", function () {
    pm.response.to.have.status(400);
});
pm.test("Verify isSuccess is false", function () {
    pm.expect(jsonData.isSuccess).to.be.false;
});
pm.test("Verify error message 'Account details not found!'", function () {
pm.expect(jsonData.errors[0]).to.eql("Account details not found!");
});
pm.test("Verify message and data are null", function () {
    pm.expect(jsonData.message).to.be.null;
    pm.expect(jsonData.data).to.be.null;
});
