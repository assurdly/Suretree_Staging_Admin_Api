pm.test("Verify that Status code is 200", function () {
    pm.response.to.have.status(200);
});
pm.test("Verify that SuccessMessage returns 'new token has been generated successfully", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData.SuccessMessage).to.eql("New token has been generated successfully.");
});
pm.test("Verify that error message returns null", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData.ErrorMessage).to.eql(null);
});
